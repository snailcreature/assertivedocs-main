typeMappingsMixin

Mixin for creating custom objects.

Source:
  • demo/customasserts.js, line 6

Example

const typeMappingsMixin = {
  object: function(arg) {
    try {
      switch (arg) {
        case "test":
          return {
            name: "John",
          }
        default:
          return {};
      }
    } catch (error) {
      return error;
    }
  }
}

module.exports = {
